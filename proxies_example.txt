# Example proxy file for user-scanner
# One proxy per line
# Supported formats:
# - http://proxy:port
# - https://proxy:port
# - socks5://proxy:port
# - proxy:port (will be treated as http://)
# Lines starting with # are ignored

# HTTP proxies
http://proxy1.example.com:8080
http://proxy2.example.com:3128

# HTTPS proxies
https://secure-proxy.example.com:443

# SOCKS5 proxies
socks5://socks-proxy.example.com:1080

# Proxies without protocol (will default to http://)
192.168.1.100:8888
10.0.0.50:3128

# Proxies with authentication
http://username:password@proxy.example.com:8080
